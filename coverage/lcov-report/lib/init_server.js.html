'use strict';
 
var appRoot = require( 'app-root-path' ).path;
 
var fs = require( 'fs' );
 
function includeRoutes( routesPath, server ){
 
	fs.readdirSync( routesPath ).forEach(function ( file ) {
 
		if ( file.substr(-3, 3) === '.js' ) {
			
			require( routesPath + "/" + file.replace('.js', '')).addRoutes( server );
		}
		
	});
 
}
 
//only to be used as a module since routes directory does not exist
function startServer( params ){
 
	var options = params.values;
 
	//import the logger
	var logger = require( './logger' );
 
	//set up the debugging
	var debug = require( 'debug' )( 'main' );
 
	//set up restify
	var restify = require( 'restify' );
	
	var routesPath = appRoot + "/routes";
 
	if( options.logging ){
 
		var server = restify.createServer( {
				name: params.name,
				log: logger,
				version: params.version
		});
 
	} 
 
	if( !options.logging ) {
 
		var server = restify.createServer( {
				name: params.name,
				version: params.version
		});
	}
 
	//basic restify configurations
	server.use( restify.queryParser() );
	server.use( restify.bodyParser({ mapParams: false }));
 
	if( options.logging ){
 
		server.pre( function (req, res, next) {  
		  
			//comment out this line to turn logging of in the console
			req.log.info({req: req}, 'START');
 
			return next();
		});
	
	}
	
	// handle all uncaught errors - make sure they fit our error format
	server.on('uncaughtException', function( req, res, route, err) {
 
		if( options.logging ) { logger.error( err, 'uncaught exception' ); }
 
		res.send( 500, { code: 500, message: (err.toString() || 'error') } );
	});
 
	includeRoutes( routesPath, server, params.routesOptions );
 
	//Run server 
	server.listen( params.port, function() {
 
		if( options.logging ) { logger.info( '%s listening at %s', server.name, server.url ) }
	});
 
}
 
module.exports = function( params ) {
 
	startServer( params );
}
