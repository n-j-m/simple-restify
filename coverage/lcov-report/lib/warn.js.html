var appRoot = require( 'app-root-path' ).path;
 
var pjson = require( appRoot + "/package.json" );
 
function warn( ){
 
	if( pjson.name === "default" ){
 
		console.warn( "@warning Edit the name of your server in package.json" );
	}
 
	if( pjson.author === "theMagicElf" ){
		
		console.warn( "@warning Edit the author of your server in package.json" );
	}
}
